```java
// UserServiceImpl.login()
public User login(String account, String password) {
    
    // ===== 校验1：账号密码非空 =====
    if (StringUtils.isBlank(account) || StringUtils.isBlank(password))
        throw new RuntimeException("账号或密码不能为空");
    
    // ===== 校验2：查数据库，支持手机号或邮箱登录 =====
    // SQL: SELECT * FROM t_user WHERE phone = ? OR email = ?
    User user = getOne(new LambdaQueryWrapper<User>()
            .eq(User::getPhone, account)
            .or()
            .eq(User::getEmail, account), false);
    
    if (user == null) throw new RuntimeException("账号不存在");
    
    // ===== 校验3：MD5 密码比对 =====
    String md5 = DigestUtils.md5DigestAsHex(password.getBytes());
    if (!md5.equals(user.getPassword())) throw new RuntimeException("密码错误");
    
    return user;  // 验证通过，返回用户对象
}
```

```java
// JwtUtil.generateMpUserToken()
public static String generateMpUserToken(Long userId, String nickname) {
    Date now = new Date();
    Date expiration = new Date(now.getTime() + 7 * 24 * 60 * 60 * 1000); // 7天
    
    return Jwts.builder()
            .setSubject(String.valueOf(userId))     // 存用户ID
            .claim("username", nickname)             // 存昵称
            .claim("pt", "MP_USER")                  // 存身份类型（关键！）
            .setIssuedAt(now)                        // 签发时间
            .setExpiration(expiration)               // 过期时间
            .signWith(SignatureAlgorithm.HS256, SECRET)  // HMAC-SHA256签名
            .compact();                              // 生成加密字符串
}
```

```java
// ProductServiceImpl.pageList()
public Page<Product> pageList(Integer page, Integer size, String name,
        Long categoryId, BigDecimal minPrice, BigDecimal maxPrice, String sort) {
    
    // ===== 阶段1：创建分页参数 =====
    Page<Product> pageParam = new Page<>(page, size);
    // MyBatis-Plus 会自动在执行前计算 offset:
    //   offset = (page - 1) * size
    //   limit  = size
    
    // ===== 阶段2：动态构建查询条件 =====
    LambdaQueryWrapper<Product> wrapper = new LambdaQueryWrapper<>();
    
    // 条件1：商品名称模糊搜索
    // SQL: WHERE name LIKE '%手机%'
    if (StringUtils.isNotBlank(name)) {
        wrapper.like(Product::getName, name);
        // like() 防止 SQL 注入：MyBatis-Plus 自动使用参数化查询
    }
    
    // 条件2：分类精确筛选
    // SQL: AND category_id = ?
    if (categoryId != null) {
        wrapper.eq(Product::getCategoryId, categoryId);
    }
    
    // 条件3-4：价格区间
    // SQL: AND price >= ? AND price <= ?
    if (minPrice != null) wrapper.ge(Product::getPrice, minPrice);
    if (maxPrice != null) wrapper.le(Product::getPrice, maxPrice);
    
    // ===== 阶段3：动态排序 =====
    if (sort == null || "default".equals(sort)) {
        // 综合排序：销量↓ → 评分↓ → 评价数↓
        wrapper.orderByDesc(Product::getSalesCount);
        wrapper.orderByDesc(Product::getAvgRating);
        wrapper.orderByDesc(Product::getReviewCount);
    } else if ("price_asc".equals(sort)) {
        wrapper.orderByAsc(Product::getPrice);   // ORDER BY price ASC
    } else if ("price_desc".equals(sort)) {
        wrapper.orderByDesc(Product::getPrice);  // ORDER BY price DESC
    } else if ("sales".equals(sort)) {
        wrapper.orderByDesc(Product::getSalesCount);
    }
    
    // ===== 阶段4：执行分页查询 =====
    // 自动执行两条 SQL：
    //   ① SELECT COUNT(*) FROM t_product WHERE ...        → pageResult.total
    //   ② SELECT * FROM t_product WHERE ... ORDER BY ... LIMIT ?, ?  → pageResult.records
    return page(pageParam, wrapper);
}
```

**LambdaQueryWrapper 的作用**：
- `Product::getName` 是 Java 方法引用，MyBatis-Plus 自动驼峰转下划线 → 数据库字段 `name`
- 所有条件都是参数化查询（`?` 占位符），**天然防 SQL 注入**
- 动态拼接：前端传什么参数就加什么条件，不传就不加

```java
// OrderServiceImpl.createOrder()

@Transactional(rollbackFor = Exception.class)  // ← 事务注解
public Order createOrder(Long userId, List<Map<String, Object>> items,
                         Long addressId,
                         String receiverName, String receiverPhone, String receiverAddress) {
    
    // ===== 阶段1：校验商品 =====
    BigDecimal totalAmount = BigDecimal.ZERO;
    for (Map<String, Object> item : items) {
        Long productId = Long.valueOf(item.get("productId").toString());
        Integer quantity = Integer.valueOf(item.get("quantity").toString());
        
        // 调 Mapper 查数据库：SELECT * FROM t_product WHERE id = ?
        Product product = productService.getById(productId);
        
        // 校验1：商品是否存在
        if (product == null) throw new RuntimeException("商品不存在");
        // 校验2：库存是否足够
        if (product.getStock() < quantity) throw new RuntimeException("库存不足");
        
        totalAmount = totalAmount.add(product.getPrice().multiply(new BigDecimal(quantity)));
    }
    
    // ===== 阶段2：获取收货地址（地址快照） =====
    String rn = receiverName;
    String rp = receiverPhone;
    String ra = receiverAddress;
    if (addressId != null) {
        // SQL: SELECT * FROM t_user_address WHERE id = ? AND user_id = ?
        UserAddress ua = userAddressService.requireOwned(userId, addressId);
        rn = ua.getReceiverName();
        rp = ua.getPhone();
        ra = buildConcatAddress(ua);  // 拼接："福建省 厦门市 集美区 XX路XX号"
        // ← 地址以快照形式保存，后续用户修改地址不影响历史订单
    }
    
    // ===== 阶段3：插入订单主表 t_order =====
    Order order = new Order();
    order.setOrderNo(generateOrderNo());    // "ORD20250621143022A1B2C3"
    order.setUserId(userId);
    order.setTotalAmount(totalAmount);
    order.setStatus(0);  // 0=待支付, 1=已支付, 2=已发货, 3=已完成, 4=已取消
    order.setReceiverName(rn);
    order.setReceiverPhone(rp);
    order.setReceiverAddress(ra);
    order.setCreateTime(LocalDateTime.now());
    
    save(order);  // ← MyBatis-Plus: INSERT INTO t_order ...
                   // save() 自动回填自增主键 order.id
    
    // ===== 阶段4：插入订单明细 + 扣减库存（同一事务） =====
    for (int k = 0; k < items.size(); k++) {
        Product product = validated.get(k);
        int quantity = qty[k];
        
        // 4.1 插入订单项
        OrderItem line = new OrderItem();
        line.setOrderId(order.getId());         // 关联订单ID
        line.setProductId(product.getId());     // 商品ID
        line.setProductName(product.getName()); // 商品名快照
        line.setProductImage(product.getMainImage()); // 商品图快照
        line.setPrice(product.getPrice());      // 价格快照
        line.setQuantity(quantity);             // 购买数量
        orderItemMapper.insert(line);
        // → SQL: INSERT INTO t_order_item (order_id, product_id, ...) VALUES (?, ?, ...)
        
        // 4.2 扣减库存
        product.setStock(product.getStock() - quantity);
        productService.updateById(product);
        // → SQL: UPDATE t_product SET stock = ? WHERE id = ?
    }
    
    // ===== 阶段5：清除购物车中已购商品 =====
    try {
        for (Map<String, Object> item : items) {
            Long productId = Long.valueOf(item.get("productId").toString());
            cartService.removeFromCart(userId, productId);
            // → SQL: DELETE FROM t_cart WHERE user_id = ? AND product_id = ?
        }
    } catch (Exception ignored) {
        // 购物车清除失败不影响订单创建
    }
    
    // ===== 阶段6：发送 RabbitMQ 延迟消息（15分钟后自动取消） =====
    try {
        orderTimeoutProducer.sendTimeout(order.getId(), userId);
    } catch (Exception ignored) {
        // 发送失败不影响下单流程
    }
    
    return order;
}  // ← @Transactional 结束，所有 SQL 一并提交或回滚
```

```java
public void handleOrderTimeout(OrderTimeoutMessage message) {
    Long orderId = message.getOrderId();
    boolean cancelled = orderService.cancelOrder(orderId);
    // cancelOrder() 内部：
    //   → SELECT * FROM t_order WHERE id = ?
    //   → 检查 status == 0（待支付）→ UPDATE t_order SET status = 4

    //   → status != 0（已支付/已取消）→ 忽略
```

```
下单
  │
  └→ orderTimeoutProducer.sendTimeout()
       │
       └→ 消息进入 order.delay.exchange
            │
            └→ order.delay.queue（TTL=900s）
                 │
                 ├─ 15分钟内支付 → status变为1，消息到期后Consumer检查status≠0，忽略
                 │
                 └─ 15分钟内未支付 → TTL到期，消息死信路由

                           │
                           └→ order.timeout.queue
```

```java
// AiChatServiceImpl.streamChat()
@Async  // ← 异步执行，不阻塞主线程
public void streamChat(Long userId, Long sessionId, String text,
                       String model, MultipartFile image, SseEmitter emitter) {
    try {
        // ===== 阶段1：保存用户消息到数据库 =====
        AiMessage userMsg = new AiMessage();
        userMsg.setSessionId(sessionId);
        userMsg.setRole("user");        // "user" 或 "assistant"
        userMsg.setContent(text);       // 用户输入的问题
        userMsg.setCreatedAt(LocalDateTime.now());
        messageMapper.insert(userMsg);
        // → SQL: INSERT INTO t_ai_message (session_id, role, content, ...) VALUES (?, ?, ?, ...)
        
        // ===== 阶段2：自动生成会话标题（首次提问时） =====
        AiSession session = sessionMapper.selectById(sessionId);
        if ("新对话".equals(session.getTitle())) {
            session.setTitle(text.length() > 15 ? text.substring(0,15) + "..." : text);
            sessionMapper.updateById(session);
        }
        
        // ===== 阶段3：初始化 ChatMemory（对话记忆，最多20条） =====

                    memory.add(AiMessage.from(msg.getContent()));
```

```java
// ProductReviewServiceImpl.submitReviews()

@Transactional(rollbackFor = Exception.class)
public void submitReviews(Long userId, List<Map<String, Object>> reviews) {
    
    Set<Long> productIds = new HashSet<>();
    Long orderId = null;
    
    for (Map<String, Object> item : reviews) {
        Long orderItemId = Long.valueOf(item.get("orderItemId").toString());
        Long productId = Long.valueOf(item.get("productId").toString());
        Integer rating = Integer.valueOf(item.get("rating").toString());
        
        // ===== 校验1：评分范围 1-5 =====
        if (rating < 1 || rating > 5) throw new IllegalArgumentException("评分必须在1-5之间");
        
        // ===== 校验2：订单项确实属于该商品 =====
        // SQL: SELECT * FROM t_order_item WHERE id = ?
        OrderItem oiEntity = orderItemMapper.selectById(orderItemId);
        if (oiEntity == null || !oiEntity.getProductId().equals(productId))
            throw new IllegalArgumentException("订单项数据异常");
        
        // ===== 校验3：所有评价必须属于同一订单 =====
        if (orderId == null) {
            orderId = oiEntity.getOrderId();
        } else if (!orderId.equals(oiEntity.getOrderId())) {
            throw new IllegalArgumentException("评价必须属于同一订单");
        }
        
        // ===== 校验4：订单必须已完成（status == 3） =====
        // SQL: SELECT * FROM t_order WHERE id = ?
        Order order = orderMapper.selectById(orderId);
        if (order == null || order.getStatus() != 3)
            throw new IllegalArgumentException("订单未完成，无法评价");
        
        // ===== 校验5：订单属于当前用户 =====
        if (!order.getUserId().equals(userId))
            throw new IllegalArgumentException("无权评价该订单");
        
        // ===== 校验6：防止重复评价 =====
        // SQL: SELECT COUNT(*) FROM t_product_review WHERE order_item_id = ?
        long count = count(new LambdaQueryWrapper<ProductReview>()
                .eq(ProductReview::getOrderItemId, orderItemId));
        if (count > 0)
            throw new IllegalArgumentException("该商品已评价，请勿重复提交");
        
        // ===== 通过所有校验，插入评价 =====
        ProductReview review = new ProductReview();
        review.setOrderId(orderId);
        review.setOrderItemId(orderItemId);
        review.setProductId(productId);
        review.setUserId(userId);
        review.setRating(rating);
        review.setContent((String) item.getOrDefault("content", ""));
        review.setCreateTime(LocalDateTime.now());
        save(review);
        // → SQL: INSERT INTO t_product_review (...) VALUES (?, ?, ..., ?)
        
        productIds.add(productId);
    }
    
    // ===== 所有评价保存后，重新计算每个商品的评分 =====
    for (Long pid : productIds) {
        updateProductRating(pid);
    }
}
```

```java
// ProductReviewServiceImpl.updateProductRating()
private void updateProductRating(Long productId) {
    // ===== 查商品 =====
    Product product = productMapper.selectById(productId);
    // → SQL: SELECT * FROM t_product WHERE id = ?
    
    // ===== 查该商品所有评价 =====
    List<ProductReview> all = list(new LambdaQueryWrapper<ProductReview>()
            .eq(ProductReview::getProductId, productId));
    // → SQL: SELECT * FROM t_product_review WHERE product_id = ?
    
    int count = all.size();
    if (count > 0) {
        // ===== 加权平均 =====
        int sum = all.stream().mapToInt(ProductReview::getRating).sum();
        // 保留1位小数，四舍五入
        BigDecimal avg = BigDecimal.valueOf(sum)
                .divide(BigDecimal.valueOf(count), 1, RoundingMode.HALF_UP);
        product.setAvgRating(avg);   // 例：4.3
    } else {
        product.setAvgRating(BigDecimal.ZERO);
    }
    product.setReviewCount(count);
    
    // ===== 更新商品评分 =====
    productMapper.updateById(product);
    // → SQL: UPDATE t_product SET avg_rating = ?, review_count = ? WHERE id = ?
}
```

```java
// OrderRefundServiceImpl.applyRefund()
@Transactional
public OrderRefund applyRefund(Long userId, Long orderId, Long orderItemId,
                               Integer type, String reason, String description) {
    
    // ===== 阶段1：校验订单状态（只有已支付/已发货/已完成可售后） =====
    Order order = orderService.getById(orderId);
    if (order.getStatus() < 1 || order.getStatus() > 3)
        throw new RuntimeException("当前订单状态不可申请售后");
    
    // ===== 阶段2：防重复申请 =====
    // 整单退款 vs 单品退款互斥检查
    if (orderItemId == null) {
        // 整单退款：不能有已有售后记录
        if (baseMapper.countByOrderIdAnyStatus(orderId) > 0)
            throw new RuntimeException("该订单已申请过售后");
    } else {
        // 单品退款：不能已有整单退款
        if (baseMapper.countByOrderItemId(orderItemId) > 0)
            throw new RuntimeException("该商品已申请售后");
    }
    
    // ===== 阶段3：计算退款金额（优惠券分摊） =====
    BigDecimal refundAmount;
    if (orderItemId != null) {
        // 单品退款：商品原价 × (实付 / 原总价) → 优惠券按比例分摊
        refundAmount = itemOriginal.multiply(paidAmount)
                .divide(allOriginal, 2, RoundingMode.HALF_UP);
    } else {
        // 整单退款：退实付金额
        refundAmount = order.getTotalAmount();
    }
    
    // ===== 阶段4：创建售后单 =====
    OrderRefund refund = new OrderRefund();
    refund.setRefundNo(generateRefundNo());   // 退款单号 REFUND20250621...
    refund.setOrderId(orderId);
    refund.setUserId(userId);
    refund.setRefundAmount(refundAmount);
    refund.setType(type);       // 1=仅退款, 2=退货退款
    refund.setReason(reason);
    refund.setStatus(type == 1 ? 0 : 1);  // 仅退款→0待审核, 退货退款→1待填物流
    save(refund);
    // → SQL: INSERT INTO t_order_refund (...) VALUES (?, ?, ..., ?)
    
    return refund;
}
```

```java
@ServerEndpoint("/ws/customer-service/{token}")
public class CustomerServiceEndpoint {

    // 在线用户（userId → WebSocket Session）
    private static final Map<String, Session> USER_SESSIONS = new ConcurrentHashMap<>();
    // 在线客服
    private static final Map<String, Session> AGENT_SESSIONS = new ConcurrentHashMap<>();

    @OnOpen
    public void onOpen(Session session, @PathParam("token") String token) {
        if (token.startsWith("agent_")) {
            // 客服上线
            AGENT_SESSIONS.put(token, session);
        } else {
            // 用户上线：token = "user_123_张三" → userId=123, 昵称=张三
            USER_SESSIONS.put(userId, session);
            broadcastToAgents("用户 张三 接入咨询");  // 通知所有客服
        }
    }

    @OnMessage
    public void onMessage(String message, Session session) {
        if (isAgent) {
            // 客服→用户：查找目标用户Session → sendToSession()
            Session targetSession = USER_SESSIONS.get(targetUser);
            sendToSession(targetSession, msg);
            svc.saveMessage(targetUser, 2, content, "text");
            // → SQL: INSERT INTO t_customer_service_message (...)
        } else {
            // 用户→客服：广播给所有在线客服
            broadcastToAgents(msg);
            svc.saveMessage(userId, 1, content, "text");
        }
    }
}
```

## 答辩建议

1. **基础功能选3-4个讲**（登录、商品列表、下单、评价），每个控制在2-3分钟
2. **拓展功能重点讲AI**（最出彩）：
   - AI智能客服：LangChain4j框架 → 自动Tool调用 → 五步操作流程
   - RAG语义搜索：自然语言 → 关键词提取 → 同义词扩展 → 加权评分
   - AI图片生成/识别：Qwen3-VL多模态 → Kolors图片生成
3. **WebSocket客服**：实时通信 vs 传统HTTP的区别
4. **优惠券抽奖**：权重随机算法 + 下单时动态计算折扣
5. **仪表盘**：环比趋势算法（今日vs昨日）
6. **答辩万能句式**："这个功能从前端 Vue 发起请求，经过 JWT 拦截器鉴权，到 Spring Boot Controller 接收参数，调 Service 层处理业务逻辑，通过 MyBatis-Plus 的 LambdaQueryWrapper 构造动态 SQL，最终操作 MySQL 数据库"
